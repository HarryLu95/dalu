###### Step 4-1 ######
column_1 = dataset[0]
column_2 = dataset[1]
l = len(column_1)
X = np.empty((l, 2))
X[:, 0] = column_1
X[:, 1] = column_2

km = KMeans(n_clusters = 3, 
            init = "random", 
            n_init = 10, 
            max_iter = 500, 
            tol = 1e-04, 
            random_state = 0)
y_km = km.fit_predict(X)

###### Step 4-2 ######
fig = plt.figure(figsize = (20, 10))

fig_km = fig.add_subplot(121, projection = None)
fig_km.scatter(X[y_km == 1, 0], 
               X[y_km == 1, 1], 
               s = 50, 
               c = "red",  
               marker = "o", 
               label = "Iris-setosa")
fig_km.scatter(X[y_km == 0, 0], 
               X[y_km == 0, 1], 
               s = 50, 
               c = "blue", 
               marker = "o", 
               label = "Iris-versicolor")
fig_km.scatter(X[y_km == 2, 0], 
               X[y_km == 2, 1], 
               s = 50, 
               c = "orange", 
               marker = "o", 
               label = "Iris-virginica")
fig_km.legend(loc = "upper right")
fig_km.grid()
fig_km.set_title("K-Means")

fig_actual = fig.add_subplot(122, projection = None)
fig_actual.scatter(X[species == "Iris-setosa", 0], 
                   X[species == "Iris-setosa", 1],

                   c = "red", 
                   marker = "o",
