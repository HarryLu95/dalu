A = np.array([])
for i in range(20):
    for j in range(20): 
        new_row = np.array([i,j,j/2])
        A = np.vstack((A,new_row)) if A.size else new_row

Sigma = np.cov(A.T)
E,W = np.linalg.eig(Sigma)
# We need to sort by largest eigenvalue, since np.linalg.eig doesn't do it automatically
idx = E.argsort()[::-1]
E = E[idx]
W = W[:,idx]

scatter = go.Scatter3d(
    x = A[:,0],
    y = A[:,1],
    z = A[:,2],
    mode='markers',
    marker=dict(
        size=5,
        symbol='circle',
        opacity=0.5
    )
)



line0 = go.Scatter3d(
    x = [0,-W[0,0]*10],
    y = [0,-W[1,0]*10],
    z = [0,-W[2,0]*10],

    line = dict(width = 10)
)

    z = [0,W[2,1]*10],

An = np.array([])
for i in range(20):
    for j in range(20): 
        new_row = np.array([i,j,j/2+np.random.normal()])
        An = np.vstack((An,new_row)) if An.size else new_row
Sigma = np.cov(An.T)
E,W = np.linalg.eig(Sigma)

idx = E.argsort()[::-1]
E = E[idx]
W = W[:,idx]

projection = (An - np.mean(An,axis=0)).dot(W[:,:2])
reconstruction = projection.dot(W[:,:2].T) + np.mean(An,axis=0)

scatter0 = go.Scatter3d(
    x = An[:,0],
    y = An[:,1],
    z = An[:,2],
    mode='markers',
    marker=dict(
        size=5,
        symbol='circle',
        opacity=0.5
    )
)

scatter1 = go.Scatter3d(
    x = reconstruction[:,0],
    y = reconstruction[:,1],
    z = reconstruction[:,2],
    mode='markers',
    marker=dict(
        size=5,
        symbol='circle',
        opacity=0.5
    )
)

data = [scatter0]
iplot(data, filename='fig1')
